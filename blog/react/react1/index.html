先把代码贴上来，在分析知识点

import React ,{Component,Fragment} from 'react'

class TodoList extends Component {
  constructor (props) {
    super(props)
    this.state = {
      value: '',
      lists: []
    }
  }

  // 输入框改变触发的函数
  handleValueChange = (e) => {
    this.setState({
      value: e.target.value
    })
  }

  // 点击添加按钮
  handleBtnClick = () => {
    this.setState({
      lists: [...this.state.lists, this.state.value],
      value: ''
    })
  }

  // 点击删除按钮
  handleDeleteClick = (index) => {
    let lists = [...this.state.lists]
    lists.splice(index, 1)
    this.setState({
      lists
    })
  }

  render () {
     return (
     <Fragment>
       <label htmlFor="input">请输入 </label>
       <input id="input" value={this.state.value} onChange={this.handleValueChange}/>
        <button onClick={this.handleBtnClick}>添加</button>
        <ul>
         {
           this.state.lists.map((item, index) => {
             return (
              <li key={index} onClick={() => this.handleDeleteClick(index)}
                dangerouslySetInnerHTML={{__html: item}}
              >
              </li>
             )
           })
         }
        </ul>
     </Fragment>
    )
  }
}

export default TodoList
  1. ... es6数组的扩展运算符,把数组或类数组对象展开成一系列用逗号隔开的值 (这个是深拷贝,和数组直接赋值不一样)
  2. 数组的map方法, 这个是es5的方法，这个一般都会用返回一个数组,经常用到的forEach、map、filter等。
  3. Fragment 片段,可以减少无用的标签(多余的)。
  4. dangerouslySetInnerHTML 不转义html标签和vue里面的v-html一个作用
  5. label, 点击label让后面的元素获取焦点,在react不是用for 而是用htmlFor

import React ,{Component,Fragment} from 'react'
import TodoItem from './TodoItem'

class TodoList extends Component {
  constructor (props) {
    super(props)
    // 唯一可以分配 this.state 的地方是构造函数。
    this.state = {
      value: '',
      lists: []
    }
  }

  // 输入框改变触发的函数
  handleValueChange = (e) => {
    this.setState({
      value: e.target.value
    })
  }

  // 点击添加按钮
  handleBtnClick = () => {

    // this.setState可以接受第二种方式改变值，传递一个函数返回一个对象。
    this.setState((prevState) => (
      {
        lists: [...prevState.lists, prevState.value],
        value: ''
      }
    ))
  }

  // 点击删除按钮
  handleDeleteClick = (index) => {
    console.log('111111111111111:', index)
    // this.setState可以接受第二种方式改变值，传递一个函数返回一个对象。
    this.setState((prevState) => {
      let lists = prevState.lists;
      lists.splice(index, 1);
      return {lists}
    })
  }

  render () {
     return (
     <Fragment>
       <label htmlFor="input">请输入 </label>
       <input id="input" value={this.state.value} onChange={this.handleValueChange}/>
        <button onClick={this.handleBtnClick}>添加</button>
        <ul>
         {
           this.state.lists.map((item, index) =>(
            <TodoItem deleteItem={this.handleDeleteClick}
              index={index} content={item} key={index}
            />
           ))
         }
        </ul>
     </Fragment>
    )
  }
}

export default TodoList
